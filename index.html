let moyens = 0;
let vieux = 0;
let und = 0;

while ( x < arrLen)
{

if ( myArray[x] < 20 )
{
console.log(myArray[x] + " = jeune")
jeune++;
}

else if ( 40 < myArray[x] )
{
console.log(myArray[x] + " = vieux")
